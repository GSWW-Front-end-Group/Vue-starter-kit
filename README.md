#Vue 项目开发构建目录简介

```
├── README.md
├── assets               //项目生成目录
├── gulpfile.babel.js    //项目gulp构建目录
├── gulpfile.js
├── package.json         
└── src                   //项目开发目录
    ├── components              //项目开发模块目录
    │   └── button
    │       ├── button.css        //模板样式文件
    │       └── button.vue         //模板定义文件
    ├── dist

    │   │   ├── common.css          
    │   │   ├── font
    │   │   ├── reset.css
    │   │   └── row-col.css           //栅格布局样式文件
    │   ├── img
    │   └── js
    │       └── lib                   //js组件目录
    └── index.html                    //项目布局页面
```
